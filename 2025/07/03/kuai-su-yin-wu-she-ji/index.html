根据SNP位点进行快速引物设计

概述

这是一个基于Python Tkinter开发的图形化界面工具，专门用于基因突变位点的引物设计。通过输入基因序列和突变位点信息，工具能够自动定位目标碱基并设计出合适的PCR引物对。

主打一个快速，但设计结果有待验证

功能特点

  • 用户友好界面：直观的图形化操作界面
  • 快速定位：自动识别基因序列中的突变位点
  • 智能设计：根据参数设置自动生成引物序列
  • 结果导出：一键复制引物序列到剪贴板

系统要求

  • 操作系统：Windows/Linux/macOS
  • 环境：Conda、 Python、 3.12
  • 依赖库：tkinter, scrolledtext, messagebox等

安装与启动

1. 环境准备

conda activate py312

2. 进入工作目录

cd E:\坚果云\我的坚果云\课题\Python

3. 启动工具

jupyter notebook

使用步骤

第一步：准备数据

  1. 从NCBI获取基因序列（如基因: LOC105346934），选取对应基因组版本，选取基因，下载基因序列
  2. 记录突变位点信息（如：NC_047567.1:26604856 T->A）

import tkinter as tk
from tkinter import ttk, scrolledtext, messagebox, simpledialog
class DNASequenceViewer:
    def __init__(self, root):
        self.root = root
        self.root.title("DNA序列定位工具")
        self.root.configure(bg='#f9f9f9')
        
        # 设置统一风格
        style = ttk.Style()
        style.theme_use('clam')
        style.configure('TButton', font=('Arial', 10), padding=3)
        style.configure('TLabel', font=('Arial', 10))
        style.configure('TEntry', font=('Arial', 10))
        style.configure('TLabelframe', font=('Arial', 11, 'bold'))
        
        # 序列输入区
        ttk.Label(root, text="输入FASTA格式序列:").pack(pady=(10, 2))
        self.seq_text = scrolledtext.ScrolledText(root, width=35, height=10, wrap=tk.WORD, font=('Courier New', 10))
        self.seq_text.pack(padx=10)

        # 坐标输入区
        self.pos_frame = ttk.Frame(root)
        self.pos_frame.pack(pady=10)
        ttk.Label(self.pos_frame, text="基因组位置:").pack(side=tk.LEFT, padx=5)
        self.pos_entry = ttk.Entry(self.pos_frame, width=15)
        self.pos_entry.pack(side=tk.LEFT)

        # 参数设置区
        self.param_frame = ttk.Labelframe(root, text="参数设置")
        self.param_frame.pack(pady=(5, 10), padx=10)

        # 参数输入控件
        for label_text, attr in [("理想产物长度:", 'product_size_entry'),
                                 ("大小容差:", 'tolerance_entry'),
                                 ("最小引物长度:", 'min_len_entry'),
                                 ("最大引物长度:", 'max_len_entry')]:
            frame = ttk.Frame(self.param_frame)
            frame.pack(anchor='w', padx=10, pady=2)
            ttk.Label(frame, text=label_text).pack(side=tk.LEFT)
            entry = ttk.Entry(frame, width=5)
            entry.pack(side=tk.LEFT, padx=5)
            setattr(self, attr, entry)

        # 设置默认值
        self.product_size_entry.insert(0, "500")
        self.tolerance_entry.insert(0, "200")
        self.min_len_entry.insert(0, "18")
        self.max_len_entry.insert(0, "22")

        # 按钮区
        self.btn_frame = ttk.Frame(root)
        self.btn_frame.pack(pady=10)
        ttk.Button(self.btn_frame, text="加载示例", command=self.load_example).pack(side=tk.LEFT, padx=5)
        ttk.Button(self.btn_frame, text="定位碱基", command=self.locate_base).pack(side=tk.LEFT, padx=5)
        ttk.Button(self.btn_frame, text="设计引物", command=self.design_primer).pack(side=tk.LEFT, padx=5)
        
        # 结果显示区
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack()
        
        # 反向互补序列显示区
        self.rev_comp_label = tk.Label(root, text="反向互补序列:")
        self.rev_comp_text = scrolledtext.ScrolledText(root, width=35, height=15, wrap=tk.WORD, state='disabled')
        
        # 状态变量
        self.is_reverse = False
        self.original_seq = ""
        self.rev_comp_seq = ""
        self.start_pos = 0
        self.end_pos = 0

    def get_clean_sequence(self):
        """提取并清理序列"""
        lines = self.seq_text.get(1.0, tk.END).split('\n')
        header = lines[0]
        seq = ''.join([line.strip() for line in lines[1:] if line and not line.startswith('>')])
        return header, seq.upper()
        
    def parse_coordinates(self, header):
        """智能解析基因组坐标（兼容正/反链表示法）
        
        支持的格式：
        - 正链：">ref|NC_001234|:100-200" 
        - 反链：">ref|NC_001234|:c200-100" 或 ">ref|NC_001234|:complement(100-200)"
        """
        try:
            # 处理标准反向链表示法（带c前缀）
            if ':c' in header:
                coord_part = header.split(':c')[-1].split()[0]
                start, end = map(int, coord_part.split('-'))
                return (start, end, True)  # 返回元组，第三个参数表示是否反向
            
            # 处理complement表示法
            elif 'complement(' in header.lower():
                coord_part = header.split('(')[1].split(')')[0]
                start, end = map(int, coord_part.split('-'))
                return (start, end, True)
            
            # 默认正向链
            elif ':' in header:
                coord_part = header.split(':')[-1].split()[0]
                if '-' in coord_part:  # 标准范围表示
                    start, end = map(int, coord_part.split('-'))
                    return (start, end, False)
                else:  # 单点位置
                    pos = int(coord_part)
                    return (pos, pos, False)
            
            raise ValueError("无法解析坐标信息")
        
        except Exception as e:
            raise ValueError(f"坐标解析错误: {str(e)}")

    def reverse_complement(self, seq):
        """生成反向互补序列"""
        comp = {'A': 'T', 'T': 'A', 'C': 'G', 'G': 'C', 'N': 'N'}
        return ''.join([comp.get(base, 'N') for base in seq[::-1]])

    def locate_base(self):
        try:
            header, seq = self.get_clean_sequence()
            start, end, is_reverse = self.parse_coordinates(header)
            target_pos = int(self.pos_entry.get())
            
            self.is_reverse = is_reverse
            self.original_seq = seq
            seq_length = abs(start - end) + 1
            
            # 验证位置范围
            if not (min(start, end) <= target_pos <= max(start, end)):
                raise ValueError(f"位点 {target_pos} 不在序列范围内 ({min(start, end)}-{max(start, end)})")
            
            # 计算相对位置
            if self.is_reverse:
                distance_from_end = max(start, end) - target_pos
                rel_pos = seq_length - distance_from_end - 1
                self.rev_comp_seq = self.reverse_complement(seq)
                working_seq = self.rev_comp_seq
                display_text = self.rev_comp_text
                strand = "负链"
                self.show_rev_comp()
            else:
                rel_pos = target_pos - min(start, end)
                working_seq = seq
                display_text = self.seq_text
                strand = "正链"
                self.hide_rev_comp()
                     
            if rel_pos >= len(working_seq) or rel_pos < 0:
                raise ValueError("计算错误：相对位置超出序列长度")
            
            base = working_seq[rel_pos]
            self.result_label.config(
                text=f"位点 {target_pos} ({strand}): 位置 {rel_pos+1}/{len(working_seq)} | 碱基 = {base}",
                fg="blue"
            )
            
            self.highlight_position(rel_pos, display_text)
            
        except Exception as e:
            self.result_label.config(text=f"错误: {str(e)}", fg="red")

    def highlight_position(self, rel_pos, text_widget):
        """精确定位高亮位置（考虑实际行长度）"""
        text_widget.tag_remove("highlight", 1.0, tk.END)
        
        # 获取文本内容（包含换行符）
        content = text_widget.get("1.0", "end-1c")
        lines = content.split('\n')
        
        # 跳过标题行（以>开头的行）
        data_lines = [line for line in lines if line and not line.startswith('>')]
        
        # 计算实际字符位置
        cumulative_len = 0
        found_line = 0
        found_col = 0
        
        for i, line in enumerate(data_lines):
            line_len = len(line)
            if cumulative_len + line_len > rel_pos:
                found_line = i + 1  # +1因为前面跳过了标题行
                found_col = rel_pos - cumulative_len
                break
            cumulative_len += line_len
        else:
            raise ValueError("无法定位位置：超出序列范围")
        
        # 高亮定位
        start_index = f"{found_line + 1}.{found_col}"  # +1因为tkinter行号从1开始
        end_index = f"{found_line + 1}.{found_col + 1}"
        
        text_widget.tag_add("highlight", start_index, end_index)
        text_widget.tag_config("highlight", background="yellow")
        text_widget.see(start_index)

    def format_sequence(self, seq, chars_per_line=35):
        """格式化序列为多行显示"""
        return '\n'.join(seq[i:i+chars_per_line] for i in range(0, len(seq), chars_per_line))

    def design_primer(self):
        """智能引物设计（自动选择最佳产物长度）"""
        try:
            # 参数设置
            IDEAL_PRODUCT_SIZE = int(self.product_size_entry.get())  # 理想产物大小
            SIZE_TOLERANCE = int(self.tolerance_entry.get())     # 产物大小允许波动范围
            MIN_PRIMER_LEN = int(self.min_len_entry.get())
            MAX_PRIMER_LEN = int(self.max_len_entry.get())
        except ValueError:
            messagebox.showerror("错误", "参数设置中存在非法数值！")
            return
    
        print(f"[调试] 获取参数成功：理想产物大小={IDEAL_PRODUCT_SIZE}, 容差={SIZE_TOLERANCE}, 引物长度={MIN_PRIMER_LEN}-{MAX_PRIMER_LEN}")
        
        # 调用实际引物设计函数或逻辑
        if not self.original_seq:
            messagebox.showwarning("警告", "请先定位碱基后再设计引物。")
            return
        try:
            # 获取序列信息
            header, seq = self.get_clean_sequence()
            start, end, is_reverse = self.parse_coordinates(header)
            target_pos = int(self.pos_entry.get())
            
            # 计算相对位置
            if is_reverse:
                working_seq = self.reverse_complement(seq)
                rel_pos = len(working_seq) - (start - target_pos) - 1
            else:
                working_seq = seq
                rel_pos = target_pos - start
            
            # 动态设计引物区域
            best_result = None
            best_score = -1
            
            # 尝试不同产物大小（300-700bp范围）
            for product_size in range(IDEAL_PRODUCT_SIZE - SIZE_TOLERANCE, 
                                    IDEAL_PRODUCT_SIZE + SIZE_TOLERANCE + 1, 50):
                
                half_size = product_size // 2
                region_start = max(0, rel_pos - half_size)
                region_end = min(len(working_seq), rel_pos + half_size)
                
                # 跳过无效范围
                if region_start >= region_end:
                    continue
                    
                # 生成候选引物
                fwd_candidates = self.get_primer_candidates(
                    working_seq, region_start, rel_pos, 
                    min_len=MIN_PRIMER_LEN, max_len=MAX_PRIMER_LEN
                )
                rev_candidates = self.get_primer_candidates(
                    working_seq, rel_pos, region_end,
                    min_len=MIN_PRIMER_LEN, max_len=MAX_PRIMER_LEN,
                    is_forward=False
                )
                
                # 评估最佳引物对
                current_result = self.evaluate_primer_pairs(
                    fwd_candidates, rev_candidates, 
                    ideal_size=IDEAL_PRODUCT_SIZE
                )
                
                # 保留最优结果
                if current_result and current_result["score"] > best_score:
                    best_result = current_result
                    best_score = current_result["score"]
            
            if not best_result:
                raise ValueError("未找到符合条件的引物，请尝试调整参数")
                
            # 补充：生成产物序列并加入 result
            f_start = best_result["forward"]["pos"]
            r_end = best_result["reverse"]["pos"] + best_result["reverse"]["len"]
            template_seq = self.reverse_complement(seq) if is_reverse else seq
            
            # 防止越界
            r_end = min(r_end, len(template_seq))
            f_start = max(0, f_start)
            
            # 生成产物序列
            product_seq = template_seq[f_start:r_end]
            best_result["product_seq"] = product_seq
                    
            # 显示结果
            self.show_primer_result(best_result)
            
        except Exception as e:
            self.result_label.config(text=f"引物设计错误: {str(e)}", fg="red")

    def calculate_gc(self, seq):
        """计算GC含量，返回百分比整数"""
        gc_count = seq.upper().count('G') + seq.upper().count('C')
        return round(100 * gc_count / len(seq))

    def calculate_tm(self, seq):
        """计算引物的Tm值（简单公式：Wallace规则）"""
        seq = seq.upper()
        a = seq.count('A')
        t = seq.count('T')
        g = seq.count('G')
        c = seq.count('C')
        tm = 2 * (a + t) + 4 * (g + c)
        return round(tm, 1)

    def has_homopolymer(self, seq, max_repeat=4):
        """
        检查是否存在连续重复碱基（如AAAA、CCCC）
        max_repeat: 允许的最大重复次数
        """
        for base in "ATCG":
            if base * (max_repeat + 1) in seq:
                return True
        return False

    
    def get_primer_candidates(self, seq, start, end, min_len=18, max_len=22, is_forward=True):
        """生成候选引物（可变长度）"""
        candidates = []
        
        for length in range(min_len, max_len + 1):
            for i in range(start, end - length + 1):
                primer = seq[i:i+length] if is_forward \
                        else self.reverse_complement(seq[i:i+length])
                
                # 跳过无效引物
                if len(primer) != length:
                    continue
                    
                # 计算引物参数
                gc = self.calculate_gc(primer)
                tm = self.calculate_tm(primer)
                
                # 基础筛选
                if not (40 <= gc <= 60):
                    continue
                if not (55 <= tm <= 65):
                    continue
                if self.has_homopolymer(primer):
                    continue
                    
                candidates.append({
                    "seq": primer,
                    "pos": i,
                    "len": length,
                    "gc": gc,
                    "tm": tm,
                    "end3": primer[-5:]  # 3'端序列（用于二聚体检查）
                })
        
        return candidates

    def has_dimer(self, seq1, seq2, min_match=4):
        """
        检查两个引物是否可能形成二聚体（简单3'端互补检查）
        min_match：最小连续碱基配对数
        """
        # 构建反向互补序列
        rc2 = self.reverse_complement(seq2)
    
        # 检查是否存在3'端配对
        for i in range(len(seq1) - min_match + 1):
            tail = seq1[-(i + min_match):]
            if tail in rc2:
                return True
    
        for i in range(len(rc2) - min_match + 1):
            tail = rc2[-(i + min_match):]
            if tail in seq1:
                return True
    
        return False


    def evaluate_primer_pairs(self, fwd_candidates, rev_candidates, ideal_size):
        """评估并选择最佳引物对"""
        best_pair = None
        best_score = -1
        
        for fwd in fwd_candidates:
            for rev in rev_candidates:
                # 计算实际产物大小
                product_size = rev["pos"] - fwd["pos"] + fwd["len"]
                
                # 基本筛选
                if product_size < 200 or product_size > 800:  # 放宽产物大小限制
                    continue
                    
                # 检查二聚体
                if self.has_dimer(fwd["seq"], rev["seq"]):
                    continue
                    
                # 综合评分
                size_diff = abs(product_size - ideal_size)
                tm_diff = abs(fwd["tm"] - rev["tm"])
                gc_diff = abs(fwd["gc"] - rev["gc"])
                
                # 评分标准（分值越高越好）
                score = 100 - 0.2*size_diff - 2*tm_diff - gc_diff
                
                # 更新最佳引物对
                if score > best_score:
                    best_pair = {
                        "forward": fwd,
                        "reverse": rev,
                        "product_size": product_size,
                        "score": score
                    }
                    best_score = score
                    
        return best_pair

界面如下：

第二步：输入数据

  1. 粘贴基因序列：将下载的FASTA格式基因序列粘贴到上方文本区域
  2. 输入突变位点：在指定位置输入突变信息
  3. 设置参数：根据需要调整引物设计参数

第三步：执行操作

  1. 定位碱基：点击”定位碱基“按钮确认突变位置

    程序会自动识别基因的方向，并显示再序列中的位置

  2. 设计引物：点击”设计引物“按钮生成引物序列

第四步：获取结果

点击”复制引物序列“按钮，所有引物信息将复制到剪贴板
