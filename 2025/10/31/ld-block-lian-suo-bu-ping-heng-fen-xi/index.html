使用LDBlockShow进行LD Block分析

在群体遗传学研究中，连锁不平衡（Linkage Disequilibrium, LD）分析是理解基因组结构和识别候选基因的重要工具。本文将介绍如何使用LDBlockShow工具对特定基因组区域进行LD block分析。本文详细介绍如何使用LDBlockShow工具对特定基因组区域进行连锁不平衡（LD）Block分析，涵盖从数据预处理、样本名修正、目标区域提取到最终可视化分析的完整流程。

4. LD Block分析

4.1 运行LDBlockShow

LDBlockShow -InVCF selected_snps.vcf.gz \
            -OutPut LOC105322102_selected_snps \
            -Region "NC_047567.1:32646281:32669029" \
            -SeleVar 2 \
            -MAF 0.05 \
            -OutPng \
            -OutPdf

4.2 参数详解

  • -InVCF selected_snps.vcf.gz：输入提取的目标SNP VCF文件
  • -OutPut LOC105322102_selected_snps：输出文件前缀
  • -Region "NC_047567.1:32646281:32669029"：分析区域
  • -SeleVar 2：选择算法
  • -MAF 0.05：过滤次要等位基因频率<5%的位点
  • -OutPng -OutPdf：生成PNG和PDF格式图片

4.3. 结果文件与解读

4.3.1 生成文件

  • LOC105322102_selected_snps.blocks：LD block定义文件
  • LOC105322102_selected_snps.blocks.gz：压缩的block文件
  • LOC105322102_selected_snps.png：LD block可视化图（PNG格式）
  • LOC105322102_selected_snps.pdf：LD block可视化图（PDF格式）

4.3.2 结果解读

LD block图中：

  • 颜色深浅：表示SNP对之间的LD强度（r²值）
  • Block边界：黑色三角标记表示LD block的边界
  • Block内部：高度连锁的SNP集合，通常作为一个整体遗传

#!/usr/bin/env Rscript

# 加载必要的包
library(tidyverse)
library(hexbin)
library(patchwork)

cat("=== LD分析可视化 ===\n")

# 1. 读取LD结果数据
ld_data <- read.table("full_ld_results.ld", header = TRUE, 
                     stringsAsFactors = FALSE, comment.char = "")

# 移除可能的空行
ld_data <- ld_data[complete.cases(ld_data), ]

cat("数据维度:", dim(ld_data), "\n")
cat("总SNP对数量:", format(nrow(ld_data), big.mark = ","), "\n")
cat("平均r²值:", round(mean(ld_data$R2, na.rm = TRUE), 4), "\n")

# 2. 随机抽样用于可视化
set.seed(123)
sample_size <- min(100000, nrow(ld_data))
ld_sample <- ld_data %>% sample_n(sample_size)

# 计算距离
ld_sample$distance <- abs(ld_sample$BP_B - ld_sample$BP_A)
ld_sample$distance_kb <- ld_sample$distance / 1000

# 3. 创建高质量LD衰减图
png("ld_decay_high_quality.png", width=12, height=10, units="in", res=300)

p1 <- ggplot(ld_sample, aes(x = distance_kb, y = R2)) +
  stat_binhex(bins = 80) +
  scale_fill_viridis_c(trans = "log10", name = "SNP pairs\n(log10)") +
  geom_smooth(color = "red", method = "loess", span = 0.3, 
              size = 1.5, se = TRUE, fill = "pink", alpha = 0.3) +
  labs(title = "A. LD Decay: r² vs Distance",
       x = "Distance between SNPs (kb)",
       y = expression(LD~(r^2))) +
  theme_minimal()

p2 <- ggplot(ld_sample, aes(x = R2)) +
  geom_histogram(fill = "lightblue", color = "black", bins = 50, alpha = 0.8) +
  geom_vline(aes(xintercept = mean(R2)), color = "red", 
             linetype = "dashed", size = 1) +
  labs(title = "B. Distribution of r² values",
       x = expression(r^2), y = "Frequency") +
  theme_minimal()

# 组合图形
combined_plot <- p1 / p2 + 
  plot_annotation(title = "Linkage Disequilibrium Analysis",
                  subtitle = paste("Based on", format(sample_size, big.mark = ","), "SNP pairs"))

print(combined_plot)
dev.off()

# 4. 创建距离分箱的LD衰减图
png("ld_by_distance_bins.png", width=10, height=6, units="in", res=300)

distance_breaks <- c(0, 1, 5, 10, 50, 100, 500, 1000, Inf)
distance_labels <- c("0-1", "1-5", "5-10", "10-50", "50-100", 
                    "100-500", "500-1000", ">1000")

ld_sample$distance_bin <- cut(ld_sample$distance_kb, 
                             breaks = distance_breaks,
                             labels = distance_labels)

ld_summary <- ld_sample %>%
  group_by(distance_bin) %>%
  summarise(mean_r2 = mean(R2, na.rm = TRUE),
            sd_r2 = sd(R2, na.rm = TRUE),
            n_pairs = n()) %>%
  filter(n_pairs > 10)

ggplot(ld_summary, aes(x = distance_bin, y = mean_r2)) +
  geom_col(fill = "steelblue", alpha = 0.8) +
  geom_errorbar(aes(ymin = mean_r2 - sd_r2, ymax = mean_r2 + sd_r2), 
                width = 0.2, color = "darkred") +
  geom_text(aes(label = format(n_pairs, big.mark = ",")), 
            vjust = -0.5, size = 3) +
  labs(title = "LD Decay by Distance Intervals",
       x = "Distance (kb)", y = "Mean r²") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

dev.off()

# 5. 生成统计报告
ld_stats <- data.frame(
  Statistic = c("Total SNP pairs", "Sampled pairs", "Mean r²", "Median r²",
                "Pairs with r² > 0.8", "Pairs with r² > 0.5"),
  Value = c(format(nrow(ld_data), big.mark = ","),
            format(sample_size, big.mark = ","),
            round(mean(ld_data$R2, na.rm = TRUE), 4),
            round(median(ld_data$R2, na.rm = TRUE), 4),
            sum(ld_data$R2 > 0.8, na.rm = TRUE),
            sum(ld_data$R2 > 0.5, na.rm = TRUE))
)

write.csv(ld_stats, "ld_detailed_statistics.csv", row.names = FALSE)

cat("分析完成！生成3个可视化文件\n")

library(tidyverse)

plot_ld_decay <- function(ld_file, output="ld_decay.png", nsample=1e5,
                          region=NULL) {
  message("=== 清洁版LD分析可视化 ===")
  ld_data <- read.table(ld_file, header=TRUE)
  message("数据维度: ", paste(dim(ld_data), collapse=" x "))
  message("列名: ", paste(colnames(ld_data), collapse=" "))

  # 区间筛选
  if (!is.null(region)) {
    parts <- strsplit(region, "[:]", fixed=FALSE)[[1]]
    if (length(parts) != 3) stop("区间格式错误，应为 chr:start:end")
    
    region_chr <- parts[1]
    start_pos  <- as.numeric(parts[2])
    end_pos    <- as.numeric(parts[3])
    
    ld_data <- ld_data %>%
      filter(CHR_A == region_chr & CHR_B == region_chr) %>%
      filter(BP_A >= start_pos & BP_A <= end_pos &
             BP_B >= start_pos & BP_B <= end_pos)
    
    message("筛选区间: ", region)
    message("区间 SNP对数量: ", nrow(ld_data))
  }

  # 统计信息
  message("\n=== LD统计摘要 ===")
  message("总SNP对数量: ", nrow(ld_data))
  message("平均r²值: ", round(mean(ld_data$R2, na.rm=TRUE), 4))
  message("r²值范围: ", paste(range(ld_data$R2, na.rm=TRUE), collapse=" - "))

  # 抽样
  set.seed(42)
  if (nrow(ld_data) > nsample) {
    ld_plot <- ld_data %>% sample_n(nsample)
  } else {
    ld_plot <- ld_data
  }

  # 画图
  plot <- ggplot(ld_plot, aes(x=abs(BP_B - BP_A), y=R2)) +
    geom_point(alpha=0.3, size=0.6) +
    geom_smooth(method="loess", formula=y~x, se=FALSE, linewidth=1, color="red") +
    theme_minimal(base_size=14) +
    labs(
      title="LD Decay",
      x="Distance (bp)",
      y=expression(r^2)
    )
  
  ggsave(output, plot, width=8, height=6, dpi=300)
  print(plot)
  message("LD衰减图已保存到: ", output)
}

解决方案：使用bcftools reheader修改样本名

问题2：区域格式错误

解决方案：确保区域格式为chr:start:end

#!/bin/bash

# 傻瓜式一键基因LD分析脚本
# 用法: ./auto_gene_ld_oneclick.sh <基因名>

set -e

# 配置参数
GENE_NAME=$1
VCF_FILE="/home/wanglab/Ywh/adductor_muscle_scar/Variants/NC_047567.1_results_snps.vcf.gz"
GFF_FILE="/path/to/your/genome.gff"  # 请修改为你的GFF文件路径
EXTEND_UPSTREAM=50000
EXTEND_DOWNSTREAM=50000
MAF_THRESHOLD=0.05

echo "=== 傻瓜式一键基因LD分析: $GENE_NAME ==="

# 检查输入参数
if [ -z "$GENE_NAME" ]; then
    echo "错误: 请提供基因名作为参数"
    echo "用法: $0 <基因名>"
    exit 1
fi

# 检查必要文件
if [ ! -f "$VCF_FILE" ]; then
    echo "错误: VCF文件不存在: $VCF_FILE"
    exit 1
fi

if [ ! -f "$GFF_FILE" ]; then
    echo "错误: GFF文件不存在: $GFF_FILE"
    echo "请修改脚本中的GFF_FILE路径"
    exit 1
fi

# 创建工作目录
WORK_DIR="LD_analysis_${GENE_NAME}"
mkdir -p "$WORK_DIR"
cd "$WORK_DIR"

echo "工作目录: $(pwd)"

# 步骤1: 从GFF文件自动提取基因坐标
echo "步骤1: 从GFF文件提取基因坐标..."

# 从GFF中提取基因信息（支持多种GFF格式）
GENE_INFO=$(grep -w "$GENE_NAME" "$GFF_FILE" | grep -E "(gene|mRNA|transcript)" | head -1)

if [ -z "$GENE_INFO" ]; then
    echo "错误: 在GFF文件中未找到基因 $GENE_NAME"
    echo "尝试模糊匹配..."
    GENE_INFO=$(grep "$GENE_NAME" "$GFF_FILE" | grep -E "(gene|mRNA|transcript)" | head -1)
    if [ -z "$GENE_INFO" ]; then
        echo "错误: 模糊匹配也失败，请检查基因名是否正确"
        exit 1
    fi
    echo "警告: 使用模糊匹配结果"
fi

echo "找到基因信息: $GENE_INFO"

# 解析GFF行（第1列: 染色体, 第4列: 起始, 第5列: 终止）
CHROM=$(echo "$GENE_INFO" | awk '{print $1}')
GENE_START=$(echo "$GENE_INFO" | awk '{print $4}')
GENE_END=$(echo "$GENE_INFO" | awk '{print $5}')

# 验证坐标
if [ -z "$CHROM" ] || [ -z "$GENE_START" ] || [ -z "$GENE_END" ]; then
    echo "错误: 无法从GFF行中解析坐标"
    echo "GFF行: $GENE_INFO"
    exit 1
fi

echo "✓ 成功提取基因坐标: $CHROM:$GENE_START-$GENE_END"

# 计算分析区域
REGION_START=$((GENE_START - EXTEND_UPSTREAM))
REGION_END=$((GENE_END + EXTEND_DOWNSTREAM))

# 确保起始位置不小于1
if [ $REGION_START -lt 1 ]; then
    REGION_START=1
fi

echo "分析区域: $CHROM:$REGION_START-$REGION_END"
echo "区域大小: $(( (REGION_END - REGION_START) / 1000 )) kb"

# 步骤2: 提取目标区域SNP
echo "步骤2: 提取目标区域SNP..."

# 方法1: 使用bcftools view -r
bcftools view -r "$CHROM:$REGION_START-$REGION_END" "$VCF_FILE" -Ov -o "${GENE_NAME}_region.vcf" 2>/dev/null || {
    echo "方法1失败，尝试方法2: 使用BED文件"
    echo -e "$CHROM\t$REGION_START\t$REGION_END" > "${GENE_NAME}_region.bed"
    bcftools view -T "${GENE_NAME}_region.bed" "$VCF_FILE" -Ov -o "${GENE_NAME}_region.vcf"
}

# 检查是否提取到SNP
SNP_COUNT=$(grep -v "^#" "${GENE_NAME}_region.vcf" | wc -l)
echo "提取到 $SNP_COUNT 个SNP"

if [ "$SNP_COUNT" -eq 0 ]; then
    echo "错误: 在目标区域未找到任何SNP"
    exit 1
fi

if [ "$SNP_COUNT" -lt 10 ]; then
    echo "警告: SNP数量较少($SNP_COUNT)，LD分析可能不理想"
fi

# 步骤3: 样本名处理
echo "步骤3: 处理样本名..."
bcftools query -l "${GENE_NAME}_region.vcf" > original_samples.txt
if [ $? -ne 0 ]; then
    echo "错误: 无法读取样本名，VCF文件可能损坏"
    exit 1
fi

sed 's/_/-/g' original_samples.txt > new_samples.txt
bcftools reheader -s new_samples.txt "${GENE_NAME}_region.vcf" -o "${GENE_NAME}_region_fixed.vcf"

# 步骤4: 转换为压缩格式
echo "步骤4: 准备分析文件..."
bgzip -f "${GENE_NAME}_region_fixed.vcf"
tabix -p vcf "${GENE_NAME}_region_fixed.vcf.gz"

# 步骤5: 运行LDBlockShow
echo "步骤5: 进行LD Block分析..."
LDBlockShow -InVCF "${GENE_NAME}_region_fixed.vcf.gz" \
            -OutPut "${GENE_NAME}_ld_block" \
            -Region "$CHROM:$REGION_START:$REGION_END" \
            -SeleVar 2 \
            -MAF $MAF_THRESHOLD \
            -OutPng \
            -OutPdf 2>&1 | grep -v "Warning" || {
    echo "LDBlockShow完成（可能有一些警告信息）"
}

# 步骤6: PLINK补充分析
echo "步骤6: 进行PLINK LD分析..."
plink --vcf "${GENE_NAME}_region_fixed.vcf.gz" \
      --double-id \
      --make-bed \
      --out "${GENE_NAME}_plink" --silent

plink --bfile "${GENE_NAME}_plink" \
      --r2 \
      --ld-window 100 \
      --ld-window-kb 1000 \
      --ld-window-r2 0 \
      --out "${GENE_NAME}_pairwise_ld" --silent

# 步骤7: 生成统计报告
echo "步骤7: 生成统计报告..."
Rscript - <<EOF

    cat("LD强度分布:\n")
    cat("  强LD(r²>0.8): ", sum(ld_data\$R2 > 0.8, na.rm = TRUE), 
        "(", round(sum(ld_data\$R2 > 0.8, na.rm = TRUE)/total_pairs*100, 1), "%)\n")
    cat("  中等LD(r²>0.5): ", sum(ld_data\$R2 > 0.5, na.rm = TRUE), 
        "(", round(sum(ld_data\$R2 > 0.5, na.rm = TRUE)/total_pairs*100, 1), "%)\n")
    cat("  弱LD(r²<0.1): ", sum(ld_data\$R2 < 0.1, na.rm = TRUE), 
        "(", round(sum(ld_data\$R2 < 0.1, na.rm = TRUE)/total_pairs*100, 1), "%)\n")
    sink()
    
    cat("✓ 统计报告生成完成\n")
} else {
    cat("警告: 未找到PLINK LD结果文件\n")
}
EOF

# 步骤8: 清理和总结
echo "步骤8: 清理临时文件..."
rm -f original_samples.txt new_samples.txt "${GENE_NAME}_region.vcf" "${GENE_NAME}_region.bed"

# 编辑脚本，修改这一行：
GFF_FILE="/path/to/your/genome.gff"  # 改为你实际的GFF文件路径

chmod +x auto_gene_ld_oneclick.sh

3. 一键运行：

./auto_gene_ld_oneclick.sh LOC105322102
./auto_gene_ld_oneclick.sh LOC105325689
./auto_gene_ld_oneclick.sh 任何基因名
