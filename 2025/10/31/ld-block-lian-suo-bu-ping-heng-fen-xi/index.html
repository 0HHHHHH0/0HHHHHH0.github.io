使用LDBlockShow进行LD Block分析

在群体遗传学研究中，连锁不平衡（Linkage Disequilibrium, LD）分析是理解基因组结构和识别候选基因的重要工具。本文将介绍如何使用LDBlockShow工具对特定基因组区域进行LD block分析。本文详细介绍如何使用LDBlockShow工具对特定基因组区域进行连锁不平衡（LD）Block分析，涵盖从数据预处理、样本名修正、目标区域提取到最终可视化分析的完整流程。

1. 分析背景与工具准备

1.1 分析目标

对基因LOC105322102区域（NC_047567.1:32646664-32667511）进行LD Block分析，了解该区域的连锁不平衡结构。

1.2 所需工具安装

# 生物信息学工具
conda install -c bioconda ldblockshow bcftools tabix vcftools plink

# R包（用于可视化）
R -e "install.packages(c('tidyverse', 'hexbin', 'patchwork', 'viridis'))"

conda activate vcftools_env

2. 数据预处理

2.1 原始数据

  • 输入文件：/home/wanglab/Ywh/adductor_muscle_scar/Variants/NC_047567.1_results_snps.vcf.gz

2.2 解决样本名问题

原始VCF文件中的样本名包含下划线，会导致PLINK（LDBlockShow内部调用）报错：

# 导出样本名并替换下划线
bcftools query -l /home/wanglab/Ywh/adductor_muscle_scar/Variants/NC_047567.1_results_snps.vcf.gz > samples.txt
sed 's/_/-/g' samples.txt > new_samples.txt

# 重新写入header生成修正后的VCF文件
bcftools reheader -s new_samples.txt /home/wanglab/Ywh/adductor_muscle_scar/Variants/NC_047567.1_results_snps.vcf.gz -o fixed_samples.vcf.gz
tabix -p vcf fixed_samples.vcf.gz

3. 目标SNP位点提取

3.1 创建目标位点列表

创建包含108个目标SNP位点的列表文件：

cat > selected_positions.txt << EOF
32667511
32661831
32661817
32661895
...（此处省略具体位点列表）
32646664
EOF

3.2 提取目标SNP

# 转换为BED格式
awk '{print "NC_047567.1\t"$1-1"\t"$1}' selected_positions.txt > selected_snps.bed

# 从修正后的VCF文件中提取目标SNP
bcftools view -T selected_snps.bed fixed_samples.vcf.gz -Oz -o selected_snps.vcf.gz
tabix -p vcf selected_snps.vcf.gz

4. LD Block分析

4.1 运行LDBlockShow

LDBlockShow -InVCF selected_snps.vcf.gz \
            -OutPut LOC105322102_selected_snps \
            -Region "NC_047567.1:32646281:32669029" \
            -SeleVar 2 \
            -MAF 0.05 \
            -OutPng \
            -OutPdf

4.2 参数详解

  • -InVCF selected_snps.vcf.gz：输入提取的目标SNP VCF文件
  • -OutPut LOC105322102_selected_snps：输出文件前缀
  • -Region "NC_047567.1:32646281:32669029"：分析区域
  • -SeleVar 2：选择算法
  • -MAF 0.05：过滤次要等位基因频率<5%的位点
  • -OutPng -OutPdf：生成PNG和PDF格式图片

4.3. 结果文件与解读

4.3.1 生成文件

  • LOC105322102_selected_snps.blocks：LD block定义文件
  • LOC105322102_selected_snps.blocks.gz：压缩的block文件
  • LOC105322102_selected_snps.png：LD block可视化图（PNG格式）
  • LOC105322102_selected_snps.pdf：LD block可视化图（PDF格式）

4.3.2 结果解读

LD block图中：

  • 颜色深浅：表示SNP对之间的LD强度（r²值）
  • Block边界：黑色三角标记表示LD block的边界
  • Block内部：高度连锁的SNP集合，通常作为一个整体遗传

5. 全基因组LD分析可视化（R脚本）

对于更全面的LD分析，可以使用以下R脚本进行可视化：

5.1 准备LD数据

# 使用PLINK计算全基因组LD
plink --vcf fixed_samples.vcf.gz \
      --double-id \
      --make-bed \
      --out full_data
      
plink --bfile full_data \
      --r2 \
      --ld-window 100 \
      --ld-window-kb 10000 \
      --ld-window-r2 0 \
      --out full_ld_results

5.2 R可视化脚本

conda activate py310

jupyter notebook --allow-root

创建 ld_visualization.R 文件：

#!/usr/bin/env Rscript

# 加载必要的包
library(tidyverse)
library(hexbin)
library(patchwork)

cat("=== LD分析可视化 ===\n")

# 1. 读取LD结果数据
ld_data <- read.table("full_ld_results.ld", header = TRUE, 
                     stringsAsFactors = FALSE, comment.char = "")

# 移除可能的空行
ld_data <- ld_data[complete.cases(ld_data), ]

cat("数据维度:", dim(ld_data), "\n")
cat("总SNP对数量:", format(nrow(ld_data), big.mark = ","), "\n")
cat("平均r²值:", round(mean(ld_data$R2, na.rm = TRUE), 4), "\n")

# 2. 随机抽样用于可视化
set.seed(123)
sample_size <- min(100000, nrow(ld_data))
ld_sample <- ld_data %>% sample_n(sample_size)

# 计算距离
ld_sample$distance <- abs(ld_sample$BP_B - ld_sample$BP_A)
ld_sample$distance_kb <- ld_sample$distance / 1000

# 3. 创建高质量LD衰减图
png("ld_decay_high_quality.png", width=12, height=10, units="in", res=300)

p1 <- ggplot(ld_sample, aes(x = distance_kb, y = R2)) +
  stat_binhex(bins = 80) +
  scale_fill_viridis_c(trans = "log10", name = "SNP pairs\n(log10)") +
  geom_smooth(color = "red", method = "loess", span = 0.3, 
              size = 1.5, se = TRUE, fill = "pink", alpha = 0.3) +
  labs(title = "A. LD Decay: r² vs Distance",
       x = "Distance between SNPs (kb)",
       y = expression(LD~(r^2))) +
  theme_minimal()

p2 <- ggplot(ld_sample, aes(x = R2)) +
  geom_histogram(fill = "lightblue", color = "black", bins = 50, alpha = 0.8) +
  geom_vline(aes(xintercept = mean(R2)), color = "red", 
             linetype = "dashed", size = 1) +
  labs(title = "B. Distribution of r² values",
       x = expression(r^2), y = "Frequency") +
  theme_minimal()

# 组合图形
combined_plot <- p1 / p2 + 
  plot_annotation(title = "Linkage Disequilibrium Analysis",
                  subtitle = paste("Based on", format(sample_size, big.mark = ","), "SNP pairs"))

print(combined_plot)
dev.off()

# 4. 创建距离分箱的LD衰减图
png("ld_by_distance_bins.png", width=10, height=6, units="in", res=300)

distance_breaks <- c(0, 1, 5, 10, 50, 100, 500, 1000, Inf)
distance_labels <- c("0-1", "1-5", "5-10", "10-50", "50-100", 
                    "100-500", "500-1000", ">1000")

ld_sample$distance_bin <- cut(ld_sample$distance_kb, 
                             breaks = distance_breaks,
                             labels = distance_labels)

ld_summary <- ld_sample %>%
  group_by(distance_bin) %>%
  summarise(mean_r2 = mean(R2, na.rm = TRUE),
            sd_r2 = sd(R2, na.rm = TRUE),
            n_pairs = n()) %>%
  filter(n_pairs > 10)

ggplot(ld_summary, aes(x = distance_bin, y = mean_r2)) +
  geom_col(fill = "steelblue", alpha = 0.8) +
  geom_errorbar(aes(ymin = mean_r2 - sd_r2, ymax = mean_r2 + sd_r2), 
                width = 0.2, color = "darkred") +
  geom_text(aes(label = format(n_pairs, big.mark = ",")), 
            vjust = -0.5, size = 3) +
  labs(title = "LD Decay by Distance Intervals",
       x = "Distance (kb)", y = "Mean r²") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

dev.off()

# 5. 生成统计报告
ld_stats <- data.frame(
  Statistic = c("Total SNP pairs", "Sampled pairs", "Mean r²", "Median r²",
                "Pairs with r² > 0.8", "Pairs with r² > 0.5"),
  Value = c(format(nrow(ld_data), big.mark = ","),
            format(sample_size, big.mark = ","),
            round(mean(ld_data$R2, na.rm = TRUE), 4),
            round(median(ld_data$R2, na.rm = TRUE), 4),
            sum(ld_data$R2 > 0.8, na.rm = TRUE),
            sum(ld_data$R2 > 0.5, na.rm = TRUE))
)

write.csv(ld_stats, "ld_detailed_statistics.csv", row.names = FALSE)

cat("分析完成！生成3个可视化文件\n")

chmod +x ld_visualization.R

library(tidyverse)

plot_ld_decay <- function(ld_file, output="ld_decay.png", nsample=1e5,
                          region=NULL) {
  message("=== 清洁版LD分析可视化 ===")
  ld_data <- read.table(ld_file, header=TRUE)
  message("数据维度: ", paste(dim(ld_data), collapse=" x "))
  message("列名: ", paste(colnames(ld_data), collapse=" "))

  # 区间筛选
  if (!is.null(region)) {
    parts <- strsplit(region, "[:]", fixed=FALSE)[[1]]
    if (length(parts) != 3) stop("区间格式错误，应为 chr:start:end")
    
    region_chr <- parts[1]
    start_pos  <- as.numeric(parts[2])
    end_pos    <- as.numeric(parts[3])
    
    ld_data <- ld_data %>%
      filter(CHR_A == region_chr & CHR_B == region_chr) %>%
      filter(BP_A >= start_pos & BP_A <= end_pos &
             BP_B >= start_pos & BP_B <= end_pos)
    
    message("筛选区间: ", region)
    message("区间 SNP对数量: ", nrow(ld_data))
  }

  # 统计信息
  message("\n=== LD统计摘要 ===")
  message("总SNP对数量: ", nrow(ld_data))
  message("平均r²值: ", round(mean(ld_data$R2, na.rm=TRUE), 4))
  message("r²值范围: ", paste(range(ld_data$R2, na.rm=TRUE), collapse=" - "))

  # 抽样
  set.seed(42)
  if (nrow(ld_data) > nsample) {
    ld_plot <- ld_data %>% sample_n(nsample)
  } else {
    ld_plot <- ld_data
  }

  # 画图
  plot <- ggplot(ld_plot, aes(x=abs(BP_B - BP_A), y=R2)) +
    geom_point(alpha=0.3, size=0.6) +
    geom_smooth(method="loess", formula=y~x, se=FALSE, linewidth=1, color="red") +
    theme_minimal(base_size=14) +
    labs(
      title="LD Decay",
      x="Distance (bp)",
      y=expression(r^2)
    )
  
  ggsave(output, plot, width=8, height=6, dpi=300)
  print(plot)
  message("LD衰减图已保存到: ", output)
}

7. 常见问题与解决方案

问题1：样本名包含下划线

错误信息：Multiple instances of '_' in sample ID
解决方案：使用bcftools reheader修改样本名

问题2：区域格式错误

错误信息：Para [-Region] should be [chr:Start:End]
解决方案：确保区域格式为chr:start:end

问题3：R包缺失

解决方案：

install.packages(c("tidyverse", "hexbin", "patchwork", "viridis"))

8. 技术要点总结

  1. 数据预处理是关键：确保VCF文件格式正确
  2. 精确提取目标区域：使用bcftools根据位点列表精确提取
  3. 多层次分析：结合LDBlockShow的区块分析和R的全基因组可视化
  4. 结果验证：检查生成的图片和统计文件，确保分析质量

9. 应用价值

这个完整流程可以：

  • 精细分析：对候选基因区域进行精细LD Block分析
  • 全局视角：通过全基因组LD分析了解整体遗传结构
  • 科研发表：生成高质量的发表级图片
  • 育种应用：为分子标记辅助选择提供依据

通过LDBlockShow和R的有机结合，研究人员可以从不同尺度全面理解基因组的连锁不平衡结构，为群体遗传学和育种研究提供强有力的技术支持。
